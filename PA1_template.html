Filling dataset

activity_tbl_filled <- activity_tbl %>% inner_join(steps_interval_mean)

## Joining by: "interval"

activity_tbl_filled <- activity_tbl_filled %>% mutate(steps = ifelse(is.na(steps), steps_interval_mean, steps))
